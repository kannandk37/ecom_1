    // UI updates instantly (optimistic)

  };

  // ── 7. Remove item — cart item row ────────────────────────────────────────
  const handleRemove = async (cartItemId: string) => {
    await removeFromCart(cartItemId);
    // UI updates instantly (optimistic)
    // on fail — auto rollback + toast
  };

  // ── 8. Increase quantity ──────────────────────────────────────────────────
  const handleIncrease = async (cartItemId: string, currentQty: number) => {
    await updateQuantity(cartItemId, currentQty + 1);
    // debounced API — UI updates on every tap instantly
  };

  // ── 9. Decrease quantity ──────────────────────────────────────────────────
  const handleDecrease = async (cartItemId: string, currentQty: number) => {
    await updateQuantity(cartItemId, currentQty - 1);

  const handleQuantityInput = async (cartItemId: string, value: string) => {
    const parsed = parseInt(value, 10);

import { useAddToCart } from "../../hooks/useAddToCart";
import { useToast } from "../../hooks/useToast"; // your toast hook

const ProductDetailScreen = ({ product }) => {
  const [isAuthModalOpen, setAuthModalOpen] = useState(false);
  const [quantity, setQuantity] = useState(1);
  const { showToast } = useToast();

  const {
    handleAddToCart,
    handlePostLoginAddToCart,
    isAddingToCart,
    isInCart,
    getCartQuantity,
  } = useAddToCart({
    showToast,
    onAuthRequired: () => setAuthModalOpen(true),
    onSuccess: () => setQuantity(1), // reset quantity after add
  });

  return (
    <div>
      {/* Quantity selector */}
      <div className="quantity-selector">
        <button onClick={() => setQuantity((q) => Math.max(1, q - 1))} disabled={isAddingToCart}>−</button>
        <span>{quantity}</span>
        <button onClick={() => setQuantity((q) => q + 1)} disabled={isAddingToCart}>+</button>
      </div>

      {/* Add to cart button */}
      <button
        onClick={() => handleAddToCart(product, quantity)}
        disabled={isAddingToCart || (product.stock !== undefined && product.stock <= 0)}
      >
        {product.stock !== undefined && product.stock <= 0
          ? "Out of Stock"
          : isAddingToCart
          ? "Adding..."
          : isInCart(product.id)
          ? "Go to Cart"
          : "Add to Cart"}
      </button>

      {/* Already in cart — show current quantity */}
      {isInCart(product.id) && (
        <span>{getCartQuantity(product.id)} in cart</span>
      )}

      <LogInOrSignUp
        isOpen={isAuthModalOpen}
        onClose={() => setAuthModalOpen(false)}
        onSuccess={handlePostLoginAddToCart}
      />
    </div>
  );
};








use case as hooks for Product card

import { useAddToCart } from "../../hooks/useAddToCart";
import { useToast } from "../../hooks/useToast";

const ProductCard = ({ product, onClick }) => {
  const { showToast } = useToast();
  const [isAuthModalOpen, setAuthModalOpen] = useState(false);

  const {
    handleAddToCart,
    handlePostLoginAddToCart,
    isAddingToCart,
    addedProductId,
    isInCart,
  } = useAddToCart({
    showToast,
    onAuthRequired: () => setAuthModalOpen(true),
    // no onSuccess — card stays where it is
  });

  // per-card loading — only show spinner on THIS card's button
  const isThisCardAdding = isAddingToCart && addedProductId === product.id;

  return (
    <div className="product-card" onClick={() => onClick?.(product)}>
      <img src={product.image} alt={product.title} />
      <h3>{product.title}</h3>
      <span>₹{product.price}</span>

      <button
        onClick={(e) => {
          e.stopPropagation(); // don't trigger card click
          handleAddToCart(product, 1);
        }}
        disabled={isThisCardAdding}
      >
        {isThisCardAdding
          ? "Adding..."
          : isInCart(product.id)
          ? "In Cart"
          : "Add to Cart"}
      </button>

      <LogInOrSignUp
        isOpen={isAuthModalOpen}
        onClose={() => setAuthModalOpen(false)}
        onSuccess={handlePostLoginAddToCart}
      />
    </div>
  );
};
